delimiter //
create procedure itr()
begin
   declare x int(100) default 0;
   pqr:loop
        if x >1001 then
             leave pqr;
        end if;
        if(x%100=0) then
            insert into tem values(x);
            set x=x+100;
        else
             iterate pqr;
        end if; 
   end loop;   
end; //
